\section{Loops}

\subsection{While Loops}
\question{Given a c-string with the name \texttt{str}, use a \uline{while} loop
to count the number of a's in the string.}

\begin{minted}[bgcolor=darkgray]{cpp}
int i = 0;
unsigned a_cnt = 0;
while (str[i] != '\0') {
    if (str[i] == 'a') {
        ++a_cnt;
    }
    ++i;
}
\end{minted}

\subsection{Do-While Loops}
\question{Let the user enter sentences. Output them back to the user but in
quotes. Stop when they say they are finished. Use a \uline{do-while} loop.}

\begin{minted}[bgcolor=darkgray]{cpp}
char str[256];
char choice;
do {
    std::cin.getline(str, 256);
    std::cout << '"' << str << '"' << std::endl;
    std::cout << "again? (y/n): ";
    std::cin.get(choice);
    std::cin.ignore(256);
} while (choice == 'y');
\end{minted}

\subsection{For Loops}
\begin{minipage}{\textwidth}
\question{Given an array of integers named \texttt{ints}, and an
\texttt{unsigned int} denoting the array size named \texttt{ints\_size}, square
the elements, in-place, using a \uline{for} loop.}

\begin{minted}[bgcolor=darkgray]{cpp}
for (unsigned i = 0; i < ints_size; ++i) {
    ints[i] = ints[i] * ints[i];
}
\end{minted}
\end{minipage}
